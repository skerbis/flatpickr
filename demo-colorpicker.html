Usage Example

// Basic usage
import colorpicker from 'pickit/colorpicker';

const picker = colorpicker('#myInput', {
  defaultColor: '#3b82f6',
  format: 'hex',
  onChange: (color) => {
    console.log('Selected color:', color);
  }
});

// With custom presets and alpha
const picker2 = colorpicker('#brandColors', {
  format: 'hsl',
  showAlpha: true,
  presetColors: [
    '#6366f1', // Indigo
    '#8b5cf6', // Purple
    '#ec4899', // Pink
    '#f59e0b', // Amber
  ],
  onChange: (color) => {
    document.body.style.backgroundColor = color;
  }
});

// Inline mode
const picker3 = colorpicker('#inlineInput', {
  inline: true,
  appendTo: document.querySelector('.color-container')
});
